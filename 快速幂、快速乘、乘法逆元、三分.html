快速幂、快速乘

LL qc(LL a, LL k, LL p) // 快速乘
{
	LL res = 0;
	while (k)
	{
		if (k & 1)
			res = (res + a) % p;
		a = (a + a) % p;
		k >>= 1;
	}
	return res;
}

LL qmi(LL a, LL k, LL p) // 快速幂
{
	LL res = 1;
	while (k)
	{
		if (k & 1)
			res = qc(res, a, p);
		a = qc(a, a, p);
		k >>= 1;
	}
	return res;
}

三分初步

#include <bits/stdc++.h>

using namespace std;

typedef long long LL;
typedef unsigned long long ULL;
typedef pair<int, int> PII;

const int N = 15;
const double eps = 1e-6;

double a[N], l, r;
int n;

double f(double x)
{
	double res = 0;
	for (int i = 0; i <= n; i ++)
		res += pow(x, i) * a[i];
	return res;
}

int main()
{
	ios::sync_with_stdio(false), cin.tie(0), cout.tie(0);
	cin >> n >> l >> r;

	for (int i = n; i >= 0; i --) cin >> a[i];

	while (r - l > eps)
	{
		double midl = l + (r - l) / 3;
		double midr = l + (r - l) / 3 * 2;
		if (f(midl) > f(midr)) r = midr;
		else l = midl;
	}
	
	cout.setf(ios::fixed);
	cout << fixed << setprecision(6) << l << endl;
	
	return 0;
}
