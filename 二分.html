整数二分模板一

情况1：要找的答案是前端的后端点

int bm(int l, int r)
{
	int ans = l;
	while(l <= r)
	{
		int mid = (l + r) / 2;//数学写法
		//int mid = l + r >> 1;//二进制写法
		if(check(mid))
		{
			ans = mid;//mid可能是答案，更新ans
			l = mid + 1;
		}
		else
		{
			r = mid - 1;
		}
	}
	return ans;
}

int bm(int l, int r)
{
	while(l < r)
	{
		int mid = (l + r + 1) / 2;//数学写法
		//int mid = l + r + 1 >> 1;//二进制写法
		if(check(mid))
		{
			l = mid;
		}
		else
		{
			r = mid - 1;
		}
	}
	return l;
	//return r;//l==r,返回l或r都行
}
