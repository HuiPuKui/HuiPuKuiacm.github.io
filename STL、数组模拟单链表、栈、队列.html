数组模拟单链表

#include <bits/stdc++.h>

using namespace std;

const int N = 100010;

int e[N], ne[N],head,idx;

void init() // 链表初始化
{
    head = -1, idx = 0;
}

void push_head(int x) // 在头结点插入x
{
    e[idx] = x;
    ne[idx] = head;
    head = idx ++;
}

void push_k(int k, int x) // 在下标为k的点后面插入x
{
    e[idx]=x;
    ne[idx]=ne[k];
    ne[k]=idx ++;
}

void remove(int k) // 删除下标是k后面的点
{
    ne[k] = ne[ne[k]];
}

int main()
{
    init();
    int m;
    cin >> m;
    while(m --)
    {
        char ch;
        int x, k;
        cin >> ch;
        if (ch == 'H')
        {
            cin >> x;
            push_head(x);
        }
        else if (ch == 'D')
        {
            cin >> k;
            if(!k)
                head = ne[head];
            else
                remove(k - 1);
        }
        else
        {
            cin >> k >> x;
            push_k(k - 1, x);
        }
    }
    
    for (int i = head; i != -1; i = ne[i]) cout << e[i] << ' ';
    cout << endl;

    return 0;
}

数组模拟栈

#include <bits/stdc++.h>

using namespace std;

const int N = 100010;

int a[N];
int tt = 0;

int main()
{
    int m;
    cin >> m;
    while(m --)
    {
        string s;
        cin >> s;
        if (s == "push") // 入栈
        {
            int x;
            cin >> x;
            a[++ tt] = x;
        }
        else if (s == "pop") // 出栈
        {
            tt --;
        }
        else if (s == "empty") // 判断栈是否为空
        {
            if(tt > 0)
                cout << "NO" << endl;
            else
                cout << "YES" << endl;
        }
        else // 输出栈顶元素
        {
            cout << a[tt] << endl;
        }
    }

    return 0;
}

数组模拟队列

#include <bits/stdc++.h>

using namespace std;

const int N = 100010;

int a[N], ed = 0, st = -1;

int main()
{
    int m;
    cin >> m;
    while(m --)
    {
        string s;
        cin >> s;
        if (s == "push") // 入队
        {
            int x;
            cin >> x;
            a[++ st] = x;
        }
        else if (s == "pop") // 出队
        {
            ed ++;
        }
        else if (s == "empty") // 判断队列是否为空
        {
            if (st < ed)
            {
                cout << "YES" << endl;
            }
            else
            {
                cout << "NO" << endl;
            }
        }
        else if (s == "query") // 输出队头元素
        {
            cout << a[ed] << endl;
        }
    }

    return 0;
}
