const int N = 1005;
int a[N][N], b[N][N];

void solve()
{
    memset(a, 0, sizeof(a));
    memset(b, 0, sizeof(b));
    int n, m, q;
    scanf("%d%d%d", &n, &m, &q);
    
    for(int i = 1; i <= n; i ++)
        for(int j = 1;j <= m; j ++)
            scanf("%d", &a[i][j]);
 
    for(int i = 1; i <= n; i ++)
        for(int j = 1; j <= m; j ++)
            b[i][j] = b[i - 1][j] + b[i][j - 1] - b[i - 1][j - 1] + a[i][j];

    while(q --)
    {
        int x1, y1, x2, y2;
        scanf("%d%d%d%d", &x1, &y1, &x2, &y2);
        printf("%d\n",b[x2][y2] - b[x2][y1 - 1] - b[x1 - 1][y2] + b[x1 - 1][y1 - 1]);
    }
}

const int N = 100005;
int a[N], b[N];

void insert(int l, int r, int k)
{
    b[l] += k;
    b[r + 1] -= k;
}

void solve()
{
    memset(a, 0, sizeof(a));
    memset(b, 0, sizeof(b));
    int n, q;
    scanf("%d%d", &n, &q);
    
    for(int i = 1; i <= n; i ++) scanf("%d", &a[i]);

    for(int i = 1; i <= n; i ++) insert(i, i, a[i]);

    while(q --)
    {
        int l, r, k;
        scanf("%d%d%d", &l, &r, &k);
        insert(l, r, k);
    }
    
    for(int i = 1; i <= n; i ++) a[i] = a[i - 1] + b[i];

    for(int i = 1; i <= n; i ++) printf("%d ", a[i]);
    printf("\n");
    
}

const int N = 1005;
int a[N][N], b[N][N];

void insert(int x1, int y1, int x2, int y2, int k)
{
    b[x1][y1] += k;
    b[x1][y2 + 1] -= k;
    b[x2 + 1][y1] -= k;
    b[x2 + 1][y2 + 1] += k;
}

void solve()
{
    memset(a, 0, sizeof(a));
    memset(b, 0, sizeof(b));
    int n, m, q;
    scanf("%d%d%d", &n, &m, &q);
    
    for(int i = 1; i <= n; i ++)
        for(int j = 1; j <= m; j ++)
            scanf("%d", &a[i][j]);
            
    for(int i = 1; i <= n; i ++)
        for(int j = 1; j <= m; j ++)
            insert(i, j, i, j, a[i][j]);
            
    while(q --)
    {
        int x1, y1, x2, y2, k;
        scanf("%d%d%d%d%d", &x1, &y1, &x2, &y2, &k);
        insert(x1, y1, x2, y2, k);
    }
    
    for(int i = 1; i <= n; i ++)
        for(int j = 1; j <= m; j ++)
            a[i][j] = a[i][j - 1] + a[i - 1][j] - a[i - 1][j - 1] + b[i][j];
            
    for(int i = 1; i <= n; i ++)
    {
        for(int j = 1; j <= m; j ++)
            printf("%d ", a[i][j]);
        printf("\n");
    }
}
